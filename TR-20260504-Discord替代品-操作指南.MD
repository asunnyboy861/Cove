#### A. LiveKit语音房间管理器

```swift
import LiveKit
import Combine

final class VoiceRoomManager: ObservableObject {
    @Published var participants: [ParticipantInfo] = []
    @Published var isConnected = false
    @Published var isMuted = true
    @Published var activeSpeaker: ParticipantInfo?
    @Published var connectionState: ConnectionState = .disconnected
    
    private var room: Room?
    private var cancellables = Set<AnyCancellable>()
    
    struct ParticipantInfo: Identifiable {
        let id: String
        let name: String
        let avatarURL: URL?
        let isMuted: Bool
        let isSpeaking: Bool
        let volume: Float
    }
    
    enum ConnectionState {
        case disconnected
        case connecting
        case connected
        case reconnecting
        case error(String)
    }
    
    func joinRoom(token: String, serverURL: URL) async throws {
        room = Room()
        guard let room = room else { return }
        
        connectionState = .connecting
        room.delegate = self
        
        let connectOptions = ConnectOptions(
            autoSubscribe: true,
            publishAudio: !isMuted
        )
        
        try await room.connect(
            url: serverURL.absoluteString,
            token: token,
            connectOptions: connectOptions
        )
        
        isConnected = true
        connectionState = .connected
        
        setupAudioProcessing()
        updateParticipants()
    }
    
    func leaveRoom() async {
        await room?.disconnect()
        room = nil
        isConnected = false
        connectionState = .disconnected
        participants = []
        activeSpeaker = nil
    }
    
    func toggleMute() {
        isMuted.toggle()
        room?.localParticipant.setMicrophoneEnabled(!isMuted)
    }
    
    private func setupAudioProcessing() {
        room?.localParticipant.publishAudioMode = .talkOnly
        
        AudioManager.shared.configure {
            $0.audioSession.category = .playAndRecord
            $0.audioSession.mode = .voiceChat
            $0.audioSession.categoryOptions = [.allowBluetooth, .defaultToSpeaker]
        }
    }
    
    private func updateParticipants() {
        guard let room = room else { return }
        
        var updated: [ParticipantInfo] = []
        
        updated.append(ParticipantInfo(
            id: room.localParticipant.sid ?? "local",
            name: room.localParticipant.identity ?? "You",
            avatarURL: nil,
            isMuted: isMuted,
            isSpeaking: room.localParticipant.isSpeaking,
            volume: room.localParticipant.audioLevel
        ))
        
        for (_, remoteParticipant) in room.remoteParticipants {
            updated.append(ParticipantInfo(
                id: remoteParticipant.sid ?? remoteParticipant.identity ?? "",
                name: remoteParticipant.identity ?? "Unknown",
                avatarURL: nil,
                isMuted: !remoteParticipant.isMicrophoneEnabled(),
                isSpeaking: remoteParticipant.isSpeaking,
                volume: remoteParticipant.audioLevel
            ))
        }
        
        participants = updated
    }
}

extension VoiceRoomManager: RoomDelegate {
    func room(_ room: Room, participantDidConnect participant: RemoteParticipant) {
        updateParticipants()
    }
    
    func room(_ room: Room, participantDidDisconnect participant: RemoteParticipant) {
        updateParticipants()
    }
    
    func room(_ room: Room, didUpdate speaking: Bool, participant: Participant) {
        if speaking {
            activeSpeaker = ParticipantInfo(
                id: participant.sid ?? "",
                name: participant.identity ?? "",
                avatarURL: nil,
                isMuted: participant.isMuted,
                isSpeaking: true,
                volume: participant.audioLevel
            )
        }
        updateParticipants()
    }
    
    func room(_ room: Room, didUpdate connectionState: ConnectionState, error: Error?) {
        if let error = error {
            self.connectionState = .error(error.localizedDescription)
        } else {
            self.connectionState = connectionState
        }
    }
}
```

#### B. Signal Protocol加密层

```swift
import Foundation
import CryptoKit
import Security

final class SignalProtocol {
    private let identityKeyStore: IdentityKeyStore
    private let sessionStore: SessionStore
    private let preKeyStore: PreKeyStore
    private let signedPreKeyStore: SignedPreKeyStore
    
    struct IdentityKeyStore {
        private let keychain = KeychainHelper()
        
        func getIdentityKeyPair() throws -> (privateKey: Curve25519.KeyAgreement.PrivateKey, publicKey: Curve25519.KeyAgreement.PublicKey) {
            if let data = keychain.load(key: "cove.identity.private") {
                let privateKey = try Curve25519.KeyAgreement.PrivateKey(rawRepresentation: data)
                return (privateKey, privateKey.publicKey)
            }
            let privateKey = Curve25519.KeyAgreement.PrivateKey()
            try keychain.save(key: "cove.identity.private", data: privateKey.rawRepresentation)
            try keychain.save(key: "cove.identity.public", data: privateKey.publicKey.rawRepresentation)
            return (privateKey, privateKey.publicKey)
        }
        
        func getRegistrationId() -> UInt32 {
            if let data = keychain.load(key: "cove.registration.id") {
                return UInt32(data.withUnsafeBytes { $0.load(as: UInt32.self) })
            }
            let id: UInt32 = UInt32.random(in: 1...16380)
            var mutableId = id
            let data = Data(bytes: &mutableId, count: MemoryLayout<UInt32>.size)
            try? keychain.save(key: "cove.registration.id", data: data)
            return id
        }
    }
    
    struct EncryptedMessage {
        let ciphertext: Data
        let ephemeralPublicKey: Data
        let previousChainLength: UInt32
        let messageNumber: UInt32
    }
    
    func encrypt(plaintext: Data, for recipientId: String) throws -> EncryptedMessage {
        let session = try sessionStore.loadSession(for: recipientId)
        let ratchet = try DoubleRatchet(session: session)
        let encrypted = try ratchet.encrypt(plaintext: plaintext)
        try sessionStore.storeSession(ratchet.serialize(), for: recipientId)
        return encrypted
    }
    
    func decrypt(encrypted: EncryptedMessage, from senderId: String) throws -> Data {
        let session = try sessionStore.loadSession(for: senderId)
        let ratchet = try DoubleRatchet(session: session)
        let plaintext = try ratchet.decrypt(encrypted: encrypted)
        try sessionStore.storeSession(ratchet.serialize(), for: senderId)
        return plaintext
    }
    
    func generatePreKeys() throws -> [PreKey] {
        let identity = try identityKeyStore.getIdentityKeyPair()
        var preKeys: [PreKey] = []
        
        for i in 0..<100 {
            let keyPair = Curve25519.KeyAgreement.PrivateKey()
            let signature = try Ed25519.Signer.PrivateKey().sign(keyPair.publicKey.rawRepresentation)
            preKeys.append(PreKey(
                id: UInt32(i),
                publicKey: keyPair.publicKey.rawRepresentation,
                signature: signature
            ))
        }
        try preKeyStore.storePreKeys(preKeys)
        return preKeys
    }
}

final class KeychainHelper {
    func save(key: String, data: Data) throws {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key,
            kSecValueData as String: data,
            kSecAttrAccessible as String: kSecAttrAccessibleAfterFirstUnlockThisDeviceOnly
        ]
        SecItemDelete(query as CFDictionary)
        let status = SecItemAdd(query as CFDictionary, nil)
        guard status == errSecSuccess else {
            throw KeychainError.saveFailed(status)
        }
    }
    
    func load(key: String) -> Data? {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key,
            kSecReturnData as String: true,
            kSecMatchLimit as String: kSecMatchLimitOne
        ]
        var result: AnyObject?
        let status = SecItemCopyMatching(query as CFDictionary, &result)
        guard status == errSecSuccess else { return nil }
        return result as? Data
    }
}
```

### 3.3 编写规则

| 规则 | 说明 |
|------|------|
| 架构模式 | MVVM + Repository Pattern,View只负责UI,ViewModel处理业务逻辑 |
| 状态管理 | @StateObject/@Observable + Combine,避免单例 |
| 网络层 | async/await + URLSession,不使用第三方网络库 |
| 数据库 | SwiftData(iOS 17+),本地优先,离线可用 |
| 加密 | CryptoKit + 自实现Signal Protocol,不依赖第三方加密库 |
| 语音 | LiveKit Swift SDK(Apache2.0),唯一第三方依赖 |
| UI | 纯SwiftUI,iOS 17+最低版本 |
| 测试 | XCTest + Swift Concurrency测试 |
| 许可证合规 | 所有依赖必须Apache2.0/MIT/BSD,避免GPL/AGPL |

### 7.7 价格策略深度解读

#### 为什么不采用"完全免费"模式?

完全免费模式适用于超级简单的工具类APP(如计算器、手电筒),但Cove有以下持续成本:
- LiveKit SFU服务器:每个语音房间$0.004/分钟/参与者
- 媒体存储:每GB约$0.023/月
- 推送通知:APNs免费但需要服务器中转
- 带宽:文字+语音+文件传输

如果完全免费,要么引入广告(违背隐私承诺),要么出售数据(违背核心价值),要么倒闭(用户再次失去家园)。

#### 为什么不采用"下载前付费购买"模式?

下载前付费适用于功能简单、一次性使用的工具(如专业相机APP),但:
- 社交APP需要网络效应,付费门槛会阻止用户加入
- 用户需要先体验才能判断是否值得付费
- Discord难民已经在付费(Nitro),对再付费有抵触

#### 为什么采用"站内订阅B"而非"站内订阅A"?

站内订阅A(免费+一次性买断)适用于功能简单、无持续API成本的APP。但Cove:
- 语音SFU有持续运营成本,无法一次性买断
- 消息翻译/AI降噪等功能有API调用成本
- 服务器维护和安全更新需要持续投入

站内订阅B(免费+按月+按年+无一次性买断)是唯一可持续的模式:
- 免费版足够日常使用,降低门槛
- Plus/Pro按月/按年订阅,覆盖运营成本
- 不提供一次性买断,因为持续成本存在
- 年付折扣27%锁定用户,降低流失率

#### 定价数字选择:为什么$3.99和$7.99?

| 价格点 | 心理效应 | 对比参照 |
|--------|----------|----------|
| $3.99 | "不到4美元"比$4.00感觉便宜很多 | Discord Nitro Basic $2.99但功能少 |
| $7.99 | "不到8美元"比$8.00感觉便宜 | Discord Nitro $9.99,比它便宜20% |
| 年付$34.99 | "不到35美元/年"=每月不到$3 | Discord Nitro年付$99.99,比它便宜65% |
| 年付$69.99 | "不到70美元/年"=每月不到$6 | Discord Nitro年付$99.99,比它便宜30% |
